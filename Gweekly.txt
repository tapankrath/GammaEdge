        function updateSliderValue() {
            var slider = document.getElementById('yieldSlider');
            document.getElementById('yieldValue').innerText = slider.value + '%';
            calculateStrategy();
        }

        function calculateStrategy() {
            var capital = parseFloat(document.getElementById('capital').value);
            var targetYieldPercent = parseFloat(document.getElementById('yieldSlider').value);

            if (isNaN(capital) || capital <= 0) {
                document.getElementById('results').innerHTML = '<p style="text-align: center; font-size: 12px; color: #64748b; font-style: italic; padding: 20px;">Please enter a valid capital base.</p>';
                return;
            }

            var totalWeeklyPayout = capital * (targetYieldPercent / 100);
            var totalAnnualPayout = totalWeeklyPayout * 52;

            var fmtCompact = function(val) {
                return new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD', maximumFractionDigits: 0 }).format(val);
            };
            
            document.getElementById('payoutDisplay').innerText = fmtCompact(totalWeeklyPayout) + ' / wk';
            document.getElementById('annualizedWarning').innerText = '(' + fmtCompact(totalAnnualPayout) + ' / yr)';

            // Structural Allocation Distribution Logic Matrix
            var etfPct = 0.55 - (targetYieldPercent * 0.04);
            var stockPct = 0.35 - (targetYieldPercent * 0.01);
            var optionsPct = 0.05 + (targetYieldPercent * 0.04);
            var hedgePct = 0.05 + (targetYieldPercent * 0.01);

            var totalWeight = etfPct + stockPct + optionsPct + hedgePct;
            etfPct /= totalWeight; stockPct /= totalWeight; optionsPct /= totalWeight; hedgePct /= totalWeight;

            var etfPrincipal = capital * etfPct;
            var stockPrincipal = capital * stockPct;
            var optionsPrincipal = capital * optionsPct;
            var hedgePrincipal = capital * hedgePct;

            // Compute Target split matrix ratios
            var etfYieldShare = totalWeeklyPayout * 0.15; 
            var stockYieldShare = totalWeeklyPayout * 0.35;
            var optionsYieldShare = totalWeeklyPayout * 0.50; 

            var htmlBlock = '';

            // Card 1: Asset Allocations Summary Block Card
            htmlBlock += '<div class="card-summary">';
            htmlBlock += '  <div class="summary-grid">';
            htmlBlock += '    <div>';
            htmlBlock += '      <div class="flex-between"><span style="color: #94a3b8;">ETFs (Collateral)</span><span style="font-weight:700;" class="color-etf">' + fmtCompact(etfPrincipal) + '</span></div>';
            htmlBlock += '      <div class="progress-container"><div class="progress-bar bg-etf" style="width: ' + (etfPct * 100) + '%"></div></div>';
            htmlBlock += '    </div>';
            htmlBlock += '    <div>';
            htmlBlock += '      <div class="flex-between"><span style="color: #94a3b8;">High-Beta Stocks</span><span style="font-weight:700;" class="color-stock">' + fmtCompact(stockPrincipal) + '</span></div>';
            htmlBlock += '      <div class="progress-container"><div class="progress-bar bg-stock" style="width: ' + (stockPct * 100) + '%"></div></div>';
            htmlBlock += '    </div>';
            htmlBlock += '    <div>';
            htmlBlock += '      <div class="flex-between"><span style="color: #94a3b8;">Spreads Engine</span><span style="font-weight:700;" class="color-options">' + fmtCompact(optionsPrincipal) + '</span></div>';
            htmlBlock += '      <div class="progress-container"><div class="progress-bar bg-options" style="width: ' + (optionsPct * 100) + '%"></div></div>';
            htmlBlock += '    </div>';
            htmlBlock += '    <div>';
            htmlBlock += '      <div class="flex-between"><span style="color: #94a3b8;">Tail-Risk Hedge</span><span style="font-weight:700;" class="color-hedge">' + fmtCompact(hedgePrincipal) + '</span></div>';
            htmlBlock += '      <div class="progress-container"><div class="progress-bar bg-hedge" style="width: ' + (hedgePct * 100) + '%"></div></div>';
            htmlBlock += '    </div>';
            htmlBlock += '  </div>';
            htmlBlock += '</div>';

            // Segment 2: Tactics Playbook Execution Layout Modules
            htmlBlock += '<div class="tactics-container">';
            
            // Engine Card #1
            htmlBlock += '  <div class="tactic-card">';
            htmlBlock += '    <div class="tactic-header">';
            htmlBlock += '      <span class="tactic-title color-etf">1. Core Collateral (SPY/QQQ)</span>';
            htmlBlock += '      <div style="text-align: right;">';
            htmlBlock += '        <span style="font-size: 11px; font-family: monospace; font-weight: 700; color: #f1f5f9; display: block;">' + fmtCompact(etfPrincipal) + ' Alloc</span>';
            htmlBlock += '        <span class="color-etf" style="font-size: 10px; font-weight: 700; display: block; margin-top: 1px;">Target: +' + fmtCompact(etfYieldShare) + '/wk</span>';
            htmlBlock += '      </div>';
            htmlBlock += '    </div>';
            htmlBlock += '    <p class="tactic-body">Sell out-of-the-money weekly Covered Calls. Baseline premium anchor with low volatility risk.</p>';
            htmlBlock += '  </div>';

            // Engine Card #2
            htmlBlock += '  <div class="tactic-card">';
            htmlBlock += '    <div class="tactic-header">';
            htmlBlock += '      <span class="tactic-title color-stock">2. The Wheel Strategy (NVDA/TSLA)</span>';
            htmlBlock += '      <div style="text-align: right;">';
            htmlBlock += '        <span style="font-size: 11px; font-family: monospace; font-weight: 700; color: #f1f5f9; display: block;">' + fmtCompact(stockPrincipal) + ' Alloc</span>';
            htmlBlock += '        <span class="color-stock" style="font-size: 10px; font-weight: 700; display: block; margin-top: 1px;">Target: +' + fmtCompact(stockYieldShare) + '/wk</span>';
            htmlBlock += '      </div>';
            htmlBlock += '    </div>';
            htmlBlock += '    <p class="tactic-body">Sell 0.20-0.30 Delta weekly Cash-Secured Puts to harvest rich high-beta Implied Volatility (IV) decay.</p>';
            htmlBlock += '  </div>';

            // Engine Card #3
            htmlBlock += '  <div class="tactic-card">';
            htmlBlock += '    <div class="tactic-header">';
            htmlBlock += '      <span class="tactic-title color-options">3. Credit Spreads (SPX Index)</span>';
            htmlBlock += '      <div style="text-align: right;">';
            htmlBlock += '        <span style="font-size: 11px; font-family: monospace; font-weight: 700; color: #f1f5f9; display: block;">' + fmtCompact(optionsPrincipal) + ' Alloc</span>';
            htmlBlock += '        <span class="color-options" style="font-size: 10px; font-weight: 700; display: block; margin-top: 1px;">Target: +' + fmtCompact(optionsYieldShare) + '/wk</span>';
            htmlBlock += '      </div>';
            htmlBlock += '    </div>';
            htmlBlock += '    <p class="tactic-body">Run short-duration (0DTE-3DTE) Bull Put Spreads. High velocity theta decay; heavily drives the premium engine.</p>';
            htmlBlock += '  </div>';

            // Engine Card #4
            htmlBlock += '  <div class="tactic-card" style="border-left: 2px solid #f43f5e;">';
            htmlBlock += '    <div class="tactic-header">';
            htmlBlock += '      <span class="tactic-title color-hedge">4. Macro Insurance (VIX / OTM Puts)</span>';
            htmlBlock += '      <div style="text-align: right;">';
            htmlBlock += '        <span style="font-size: 11px; font-family: monospace; font-weight: 700; color: #cbd5e1; display: block;">' + fmtCompact(hedgePrincipal) + ' Budget</span>';
            htmlBlock += '        <span class="color-hedge" style="font-size: 10px; font-weight: 700; display: block; margin-top: 1px;">Sunk Cost</span>';
            htmlBlock += '      </div>';
            htmlBlock += '    </div>';
            htmlBlock += '    <p class="tactic-body">Buy 60-90 days out long puts. This capital does not generate yield; it acts purely as system insurance to catch an emergency downturn.</p>';
            htmlBlock += '  </div>';

            htmlBlock += '</div>';

            document.getElementById('results').innerHTML = htmlBlock;
        }

        function sharePlan() {
            var capital = document.getElementById('capital').value;
            var targetYieldPercent = document.getElementById('yieldValue').innerText;
            var weeklyPayout = document.getElementById('payoutDisplay').innerText;
            var annualPayout = document.getElementById('annualizedWarning').innerText;

            var shareText = "Alpha Yield Strategy Plan:\n" +
                            "• Capital Base: $" + parseFloat(capital).toLocaleString() + "\n" +
                            "• Yield Goal: " + targetYieldPercent + "\n" +
                            "• Expected Revenue: " + weeklyPayout + " " + annualPayout + "\n" +
                            "Generated via Alpha Yield Planner Engine.";

            if (navigator.share) {
                navigator.share({
                    title: 'Alpha Yield Strategy Allocation',
                    text: shareText
                }).catch(function(err) {
                    console.log("Share cancelled: ", err);
                });
            } else {
                navigator.clipboard.writeText(shareText);
                alert("Plan data copied to clipboard!");
            }
        }

        window.onload = function() {
            calculateStrategy();
        };
